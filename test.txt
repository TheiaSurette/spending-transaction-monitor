WITH last_txn AS (
    SELECT user_id, transaction_date, amount, merchant_name, merchant_category, trans_num,
           merchant_city, merchant_state, merchant_country, merchant_latitude, merchant_longitude, merchant_zipcode
    FROM transactions

    LIMIT 1
  ),
  historical AS (
    SELECT COALESCE(AVG(t.amount), 0) AS avg_amount
    FROM transactions t
    CROSS JOIN last_txn lt
    WHERE t.user_id = lt.user_id
      AND LOWER(t.merchant_name) LIKE LOWER(lt.merchant_name)
      AND t.transaction_date >= lt.transaction_date - INTERVAL '60 days'
      AND t.transaction_date < lt.transaction_date
  ),
  recent_txns AS (
    SELECT COUNT(*) AS count
    FROM transactions t
    CROSS JOIN last_txn lt
    WHERE t.user_id = lt.user_id
      AND LOWER(t.merchant_name) LIKE LOWER(lt.merchant_name)
      AND t.transaction_date >= lt.transaction_date - INTERVAL '60 days'
      AND t.transaction_date <= lt.transaction_date
  ),
  prior_txns AS (
    SELECT COUNT(*) AS count
    FROM transactions t
    CROSS JOIN last_txn lt
    WHERE t.user_id = lt.user_id
      AND LOWER(t.merchant_name) LIKE LOWER(lt.merchant_name)
      AND t.transaction_date < lt.transaction_date - INTERVAL '60 days'
  )
  SELECT CASE
    WHEN NOT EXISTS (SELECT 1 FROM prior_txns WHERE count > 0)
      AND (SELECT count FROM recent_txns) > 1
      AND lt.amount - (SELECT avg_amount FROM historical) > 5
      THEN 'ALERT: Recurring charge amount increased by more than $5'
    ELSE 'NO_ALERT'
  END AS alert
  FROM last_txn lt;
